//First code compiled and executed on ESP-32 dev board.
// Need to have silicon labs windows 32 driver installed below URL here on windows laptop: 
// https://www.silabs.com/software-and-tools/usb-to-uart-bridge-vcp-drivers?tab=downloads

void setup() {
  pinMode(2, OUTPUT); // Assuming the on-board LED is on pin 6
}

void loop() {
  digitalWrite(2, HIGH); // Turn the LED on
  delay(1000);           // Wait for a second
  digitalWrite(2, LOW);  // Turn the LED off
  delay(1000);           // Wait for a second
}
